def emit(data, chain)
  @top.synchronize {
    if @top.size + data.size < @chunk_limit
      chain.next
      @top << data
      return false

    elsif @queue.size >= @queue_limit
      # TODO
      raise BufferError, "queue size exceeds limit"

    else
      n = new_data
      ok = false
      begin
        n << data
        chain.next
        @queue.synchronize {
          @queue << @top
        }
        @top.close_write rescue nil
        @top = n
        ok = true
      ensure
        n.purge unless ok
      end
      return true
    end
  }
end

def shutdown
  @top.synchronize {
    @queue.synchronize {
      until @queue.empty?
        @queue.shift.close
      end
    }
    @top.close
  }
end

def try_flush(out, force)
  if @queue.empty?
    #unless force
    #  return false
    #end
    @top.synchronize {
      unless @top.empty?
        n = new_data
        ok = false
        begin
          @queue.synchronize {
            @queue << @top
          }
          @top.close_write rescue nil
          @top = n
          ok = true
        ensure
          n.purge unless ok
        end
      end
    }
  end

  chunk = nil
  @queue.synchronize {
    if @parallel
      chunk = @queue.find {|c| c.try_mon_enter }
      return false unless chunk
    else
      chunk = @queue.first
      return false unless chunk
      return false unless chunk.try_mon_enter
    end
  }

  begin
    if !chunk.empty?
      write_chunk(chunk, out)
    end

    @queue.synchronize {
      @queue.delete_if {|c|
        c.object_id == chunk.object_id
      }
    }
    chunk.purge

    return !@queue.empty?
  ensure
    chunk.mon_exit
  end
end
