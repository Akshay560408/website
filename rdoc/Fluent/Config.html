115:     attrs = {}
116:     elems = []
117:     while i < lines.length
118:       line = lines[i]
119:       line.lstrip!
120:       line.gsub!(/\s*(?:\#.*)?$/,'')
121:         if line.empty?
122:           i += 1
123:           next
124:         elsif m = /^\<([a-zA-Z0-9_]+)\s*(.+?)?\>$/.match(line)
125:           e_name = m[1]
126:           e_arg = m[2] || ""
127:           i, e_attrs, e_elems = parse_element(e_name, lines, i+1, fname)
128:           elems << Element.new(e_name, e_arg, e_attrs, e_elems)
129:         elsif line == "</#{name}>"
130:           i += 1
131:           break

134:           i += 1
